Reverse an Array in Go

// Go program to reverse an array
package main

import (
    "fmt"
)

// reverseArray reverses an array in place
func reverseArray(arr []int) {
    start, end := 0, len(arr)-1
    for start < end {
        arr[start], arr[end] = arr[end], arr[start]
        start++
        end--
    }
}

// printArray prints the array elements
func printArray(arr []int) {
    for _, num := range arr {
        fmt.Print(num, " ")
    }
    fmt.Println()
}

func main() {
    arr := []int{1, 2, 3, 4, 5}
    fmt.Println("Original Array:")
    printArray(arr)

    reverseArray(arr)

    fmt.Println("Reversed Array:")
    printArray(arr)
}
